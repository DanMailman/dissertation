% terms.tex — glossary of terms entries

\newglossaryentry{lexeme}{%
	type=terms,
	name={lexeme},
	description={approximately a word or a common set phrase in a language; used in this dissertation as the basic unit of lexical input}
}

\newglossaryentry{inputset}{%
	name={input set},
	description={a collection of language elements (e.g., characters, words, or phrases) that can be entered using an input method; examples include specific languages such as English or Chinese, and specialized systems such as the IPA}
}

\newglossaryentry{selectiongrid}{%
	name={selection grid},
	description={a display-resident grid whose cells present candidate lexemes or control items that can be chosen via keyboard, mouse, touch, or other gestures}
}

\newglossaryentry{gesture}{%
	name={gesture},
	description={a single atomic user action recognized by the input system, such as a keypress, mouse click, or tap}
}
